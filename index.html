Criar tabela de perfis para controle de acessos/funções CREATE TABLE public.perfis ( id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY, matricula TEXT UNIQUE NOT NULL, nome TEXT NOT NULL, role TEXT NOT NULL DEFAULT 'estagiario' CHECK (role IN ('estagiario', 'admin')), created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL ); -- Habilitar RLS nos Perfis ALTER TABLE public.perfis ENABLE ROW LEVEL SECURITY; CREATE POLICY "Permitir leitura pública de perfis" ON public.perfis FOR SELECT USING (true); CREATE POLICY "Admins podem atualizar perfis" ON public.perfis FOR UPDATE USING ( EXISTS (SELECT 1 FROM public.perfis WHERE id = auth.uid() AND role = 'admin') ); -- 2. Atualizar a tabela registros_ponto para suportar abonos e assinaturas do admin ALTER TABLE public.registros_ponto ADD COLUMN abonado BOOLEAN DEFAULT false, ADD COLUMN justificativa_abono TEXT, ADD COLUMN editado_por_admin_id UUID REFERENCES auth.users(id); -- 3. Criar tabela de solicitações de alteração de ponto CREATE TABLE public.solicitacoes_ponto ( id BIGSERIAL PRIMARY KEY, usuario_id UUID REFERENCES auth.users(id) ON DELETE CASCADE NOT NULL, matricula TEXT NOT NULL, nome_estagiario TEXT NOT NULL, data_ponto DATE NOT NULL, tipo_batida TEXT NOT NULL, horario_sugerido TIME NOT NULL, motivo TEXT NOT NULL, status TEXT DEFAULT 'pendente' CHECK (status IN ('pendente', 'aprovado', 'rejeitado')), resposta_admin TEXT, created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc'::text, NOW()) NOT NULL ); ALTER TABLE public.solicitacoes_ponto ENABLE ROW LEVEL SECURITY; CREATE POLICY "Estagiários veem/criam suas próprias solicitações" ON public.solicitacoes_ponto FOR ALL USING (auth.uid() = usuario_id); CREATE POLICY "Admins controlam todas as solicitações" ON public.solicitacoes_ponto FOR ALL USING (EXISTS (SELECT 1 FROM public.perfis WHERE id = auth.uid() AND role = 'admin'));
